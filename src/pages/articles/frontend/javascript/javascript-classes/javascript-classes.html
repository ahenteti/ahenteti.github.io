Before ES6 specification

            function Animal(name) {
              this.name = name;
            }
            
            Animal.prototype.getName = function() {
              return this.name;
            }
            
            const dog = new Animal('Dog');
            dog.getName();

After ES6 specification

            class Animal {
              constructor(name) {
                this.name = name;
              }
              getName() {
                return this.name;
              }
            }
            
            const dog = new Animal('Dog');
            dog.getName();

ES6 static methods

            class Animal {
              constructor(name) {
                this.name = name;
              }
              getName() {
                return this.name;
              }
              static getNames() {
                for (const animal of arguments) {
                  console.log(animal.getName());
                }
              }
            }
            
            const dog = new Animal('Dog');
            const cat = new Animal('Cat');
            Animal.getNames(dog, cat);
