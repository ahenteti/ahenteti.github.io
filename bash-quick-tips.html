How to split a string on delimiter

                
                    IFS='-' read -ra array <<< "a-b-c"; unset IFS
                    for i in ${array[@]}; do
                        echo $i
                    done

How to remove particular characters from a string

                
                    # using tr
                    echo "Hello World!" | tr -d !
    
                    # using sed
                    echo "Hello World!" | sed 's/!//g'
