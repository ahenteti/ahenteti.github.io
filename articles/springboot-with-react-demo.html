App.js 
                        
                            import React, {Component} from 'react';
                            import './App.css';
                            
                            class App extends Component {
                            
                                state = {};
                            
                                componentDidMount() {
                                    setInterval(this.hello, 1000);
                                }
                            
                                hello = () => {
                                    fetch('/api/hello')
                                        .then(response => response.text())
                                        .then(message => {
                                            this.setState({message: message});
                                        });
                                };
                            
                                render() {
                                    return (
                                        <div className="App">
                                             <h1 className="App-title">{this.state.message}</h1>
                                        </div>
                                    );
                                }
                            }
                            
                            export default App;

Usage

            
                git clone https://github.com/ahenteti/ahenteti.github.io.demo.git
                cd ahenteti.github.io.demo/spring/spring-boot-with-react-demo
                mvn clean install -Pbuild
                # then open this url http://localhost:9999 in your browser
